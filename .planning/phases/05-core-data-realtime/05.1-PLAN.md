Create `src/lib/supabaseHelpers.ts` — a thin helper layer used by all migrated stores:

```typescript
import { supabase, isSupabaseConfigured } from './supabase'
import { useAuthStore } from '../store/authStore'

/** Get current org ID from authStore — used for all inserts */
export function getOrgId(): string {
  const orgId = useAuthStore.getState().organizationId
  if (!orgId) throw new Error('[supabaseHelpers] No organizationId in authStore')
  return orgId
}

/** Generic select with org scoping (RLS handles it, but explicit for clarity) */
export async function sbSelect<T>(table: string): Promise<T[]> {
  if (!isSupabaseConfigured || !supabase) return []
  const { data, error } = await supabase.from(table).select('*').order('created_at', { ascending: false })
  if (error) throw error
  return (data ?? []) as T[]
}

/** Generic insert — adds organization_id automatically */
export async function sbInsert<T extends Record<string, unknown>>(table: string, row: T): Promise<T & { id: string }> {
  if (!isSupabaseConfigured || !supabase) throw new Error('Supabase not configured')
  const { data, error } = await supabase.from(table).insert({ ...row, organization_id: getOrgId() }).select().single()
  if (error) throw error
  return data as T & { id: string }
}

/** Generic update by id */
export async function sbUpdate<T extends Record<string, unknown>>(table: string, id: string, updates: T): Promise<void> {
  if (!isSupabaseConfigured || !supabase) throw new Error('Supabase not configured')
  const { error } = await supabase.from(table).update({ ...updates, updated_at: new Date().toISOString() }).eq('id', id)
  if (error) throw error
}

/** Generic delete by id */
export async function sbDelete(table: string, id: string): Promise<void> {
  if (!isSupabaseConfigured || !supabase) throw new Error('Supabase not configured')
  const { error } = await supabase.from(table).delete().eq('id', id)
  if (error) throw error
}

/** Generic bulk delete */
export async function sbBulkDelete(table: string, ids: string[]): Promise<void> {
  if (!isSupabaseConfigured || !supabase) throw new Error('Supabase not configured')
  const { error } = await supabase.from(table).delete().in('id', ids)
  if (error) throw error
}
```

Note: Supabase column names use snake_case (created_at, updated_at, organization_id) but the TypeScript types use camelCase (createdAt, updatedAt). You'll need to handle the mapping. Check the Contact type in src/types/index.ts to see what fields exist. The database columns are snake_case. Add a `toCamel` and `toSnake` mapper to supabaseHelpers.ts OR use Supabase's built-in column naming if available. The simplest approach: just map in each store individually since field names vary.
